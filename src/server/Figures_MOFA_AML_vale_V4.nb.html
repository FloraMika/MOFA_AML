keep_top_rows_df <- function(df,n_top=5) {
  # Create a new data frame with NA values
  df_new <- data.frame(matrix(NA, ncol = ncol(df), nrow = nrow(df)))
  rownames(df_new) <- rownames(df)
  colnames(df_new) <- colnames(df)
  
  # Loop through each column
  for (i in 1:ncol(df)) {
    # Select the top 5 rows for the current column
    top_rows <- order(df[, i], decreasing = TRUE)[1:n_top]
    
    # Copy the values from the original data frame to the new data frame
    df_new[top_rows, i] <- df[top_rows, i]
  }
  df_new <- df_new[rowSums(is.na(df_new)) != ncol(df_new), ]
  df_new[is.na(df_new)] = 0
  # Return the new data frame
  return(df_new)
}
