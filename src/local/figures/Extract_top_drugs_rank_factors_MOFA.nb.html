mofa.sel <- select_model(mofa.list)
list_drugs <- c("Venetoclax", "Azacitidine", "Fludarabine", "Decitabine", "Crenolanib", "Midostaurin", "Sorafenib", "Quizartinib", "Gilteribinib")

weights <- get_weights(mofa.sel, views = "DSRT", factor = 2, scale = TRUE, as.data.frame = TRUE)
weights <- merge(conv, weights, by = "feature")
weights <- weights[order(weights$value, decreasing = FALSE),]
weights <- weights[1:30,]
weights$feature.name <- factor(weights$feature.name, levels = rev(weights$feature.name))
data <- weights
colnames(data)[c(2, 5)] <- c("x", "y")
# Plot
p <- ggplot(data, aes(x=x, y=y)) +
