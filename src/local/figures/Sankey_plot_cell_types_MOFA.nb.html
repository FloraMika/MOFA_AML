cell_neg <- read.csv("/home/flomik/Code/MOFA_AML/data/server/cell_types_top_pathways_tojo_negative.csv", row.names = 1)
row_names <- rownames(cell_neg)
cell_neg_2  <- stack(cell_neg)
cell_neg_2$ind_2 <- rep(row_names, ncol(cell_neg))
cell_neg_2$sign <- "neg"
cell_neg_2$type <- "cell-factor"

cell_pos <- read.csv("/home/flomik/Code/MOFA_AML/data/server/cell_types_top_pathways_tojo_positive.csv", row.names = 1)
row_names <- rownames(cell_pos)
cell_pos_2  <- stack(cell_pos)
cell_pos_2$ind_2 <- rep(row_names, ncol(cell_pos))
cell_pos_2$sign <- "pos"
cell_pos_2$type <- "cell-factor"
cell_type <- rbind(cell_pos_2, cell_neg_2)
cell_type <- cell_type[!is.na(cell_type$values),]
cell_type$ind <- gsub("Factor", "DDHM", cell_type$ind)
cell_type <- cell_type[order(cell_type$sign, cell_type$values),]
cell_type$ind_2 <- factor(as.vector(cell_type$ind_2), levels = as.vector(cell_type$ind_2)[!duplicated(cell_type$ind_2)])
cell_type_filt <- cell_type[!grepl("100", cell_type$ind_2),]
cell_type_filt$Pathway <- as.vector(cell_type_filt$ind_2)
cell_type_filt$Pathway <- gsub("\\.Top250", "", cell_type_filt$Pathway)
library(ggalluvial)
cell_type_filt_pos <- cell_type_filt[cell_type_filt$sign == "pos",]
cell_type_filt_pos$values <- NULL
cell_type_filt_pos$sign <- NULL
cell_type_filt_pos$ind_2 <- NULL
cell_type_filt_pos$type <- NULL
colnames(cell_type_filt_pos)[2] <- "Cell_P"
cell_type_filt_neg <- cell_type_filt[cell_type_filt$sign == "neg",]
cell_type_filt_neg$values <- NULL
cell_type_filt_neg$sign <- NULL
cell_type_filt_neg$ind_2 <- NULL
cell_type_filt_neg$type <- NULL
colnames(cell_type_filt_neg)[2] <- "Cell_N"
data <- merge(cell_type_filt_pos, cell_type_filt_neg, by = "ind", all.x = TRUE, all.y = TRUE)
data$Freq <- 1

data$Cell_N[data$Cell_N == "MLL_LSC_Somervaille2009_Down"] <- "MLL_LSC(down)"


data$Cell_P[data$Cell_P == "LSC104_Ng2017_DOWN"] <- "LSC(down)"
