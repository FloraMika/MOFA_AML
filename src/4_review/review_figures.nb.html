comb <- list(c("Proteomics", "Factor1"),
             c("Mutations", "Factor1"),
             c("DSRT", "Factor2"),
             c("DNAm", "Factor1"),
             c("DNAm", "Factor3"))
path_factors <- "data/MOFA/AML_MOFAplus_top_factors_v04_f600 (1).xlsx"
table_score <- data.frame(patient_id = mofa.sel@samples_metadata$sample)

for(i in comb){
  mod <- i[1]
  fact <- i[2]
  print(c(fact,mod))
  
  ## extract top features for specific factor
  factorName=paste0(fact,"_", mod)
  RNAfac=read_excel(path_factors,sheet=factorName)
  RNAfac=data.table(RNAfac)
  RNAfac$feature <- gsub("\\..", "", RNAfac$feature)
  RNAfac$feature[RNAfac$feature == "FIMM000406IMM000173"] <- "FIMM000406.FIMM000173"
  print(nrow(RNAfac))
  
  ## select data
  data <- data.frame(mofa.sel@data[mod][[1]]$AML)
  emptycols <- sapply(data, function (k) all(is.na(k)))
  data <- data[!emptycols]
  rownames(data) <- gsub("\\..", "", rownames(data))
  rownames(data)[rownames(data) == "FIMM000406IMM000173"] <- "FIMM000406.FIMM000173"
  condition <- data.frame(patient_id = colnames(data))
  
  ## calculate score per patient
  pick <- rownames(data) %in% RNAfac$feature
  x=t(data[pick,drop=FALSE,])
  beta <- RNAfac$value[match(colnames(x), RNAfac$feature)]
  yval=t(t(x) * beta)
  yval=rowSums(yval)
  weight <- data.frame(yval)
  colnames(weight) <- fact
  weight$patient_id <- rownames(weight)
  yval <- data.frame(patient_id = names(yval), score = yval)
  yval$group <- as.integer(yval$score > median(yval$score))+1
  colnames(yval)[2:3] <- paste0(mod,"_",fact, "_", colnames(yval)[2:3])
  condition <- merge(condition, yval, by = "patient_id")
  table_score <- merge(table_score, condition, all.x = TRUE)
}
